23inline unsigned long long transform(unsigned long long value, int(*indexMapper)(int r, int c)) {
24 unsigned long long result = 0;
25 for (int r = 0; r < MATRIX_SIZE; r++) {
26 for (int c = 0; c < MATRIX_SIZE; c++) {
27 int flatIndex = r * MATRIX_SIZE + c;
28 int rotatedIndex = indexMapper(r, c);
29 unsigned long long rotatedIndexValue = value & (1ULL << rotatedIndex);
30 result += (1ULL << flatIndex) * (rotatedIndexValue ? 1ULL : 0ULL);
31 }
32 }
33 return result;
34}
