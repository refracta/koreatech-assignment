10int generateLeftScrollImage(DotMatrixImage images[], String text) {
11 int len = text.length();
12 int index = 0;
13 for (int i = 0; i < len; i++) {
14 unsigned long long currentFont = toSimpleFont(text.charAt(i));
15 unsigned long long nextFont = toSimpleFont(text.charAt((i + 1) % len));
16 unsigned long long doubleNextFont = toSimpleFont(text.charAt((i + 2) % len));
17 DotMatrixImage currentImage = {currentFont};
18 DotMatrixImage nextImage = {nextFont};
19 DotMatrixImage doubleNextImage = {doubleNextFont};
20
21 images[index++] = currentImage | nextImage.shift(RIGHT, 5);
22 images[index++] = currentImage.shift(LEFT, 1) | nextImage.shift(RIGHT, 4);
23 images[index++] = currentImage.shift(LEFT, 2) | nextImage.shift(RIGHT, 3);
24 images[index++] = currentImage.shift(LEFT, 3) | nextImage.shift(RIGHT, 2) | doubleNextImage.shift(RIGHT, 7);
25 images[index++] = nextImage.shift(RIGHT, 1) | doubleNextImage.shift(RIGHT, 6);
26 }
27 return index;
28}

30DotMatrixImage generateLeftScrollImage(String text, int frame) {
31 int charIndex = frame / 5;
32 int animateIndex = frame % 5;
33 int len = text.length();
34 unsigned long long currentFont = toSimpleFont(text.charAt(charIndex));
35 unsigned long long nextFont = toSimpleFont(text.charAt((charIndex + 1) % len));
36 unsigned long long doubleNextFont = toSimpleFont(text.charAt((charIndex + 2) % len));
37 DotMatrixImage currentImage = {currentFont};
38 DotMatrixImage nextImage = {nextFont};
39 DotMatrixImage doubleNextImage = {doubleNextFont};
40
41 DotMatrixImage result = nextImage.shift(RIGHT, 5 - animateIndex);
42 if (animateIndex < 4) {
43 result = result | (animateIndex == 0 ? currentImage : currentImage.shift(LEFT, animateIndex));
44 }
45 if (animateIndex > 2) {
46 result = result | doubleNextImage.shift(RIGHT, 10 - animateIndex);
47 }
48 return result;
49}

110
111 for (int c = 0; c < MATRIX_SIZE; c++) {
112 digitalWrite(RED_PINS[c], bitRead(current.redImage, c) ? LOW : HIGH);
113 digitalWrite(GREEN_PINS[c], bitRead(current.greenImage, c) ? LOW : HIGH);
114 controlRowPulse(r);

116 controlRowPulse(-1);
117 }
